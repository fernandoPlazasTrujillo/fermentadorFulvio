125void task_actuadores(void *pvParameters)
126{
127 control_cmd_t cmd;
128
129 actuadores_init();
130
131 bool motor_activo = false;
132 TickType_t motor_start_time = 0;
133
134 float last_servo_angle = -1;
135
136 ESP_LOGI(TAG, "Task actuadores iniciada");
137
138 while (1) {
139
140 if (xQueueReceive(queue_control, &cmd, pdMS_TO_TICKS(100))) {
141
142 ESP_LOGI(TAG, "Ejecutando acciones...");
143
144 gpio_set_level(PIN_BOMBA, cmd.bomba ? 1 : 0);
145
146 if (cmd.servo_angle != last_servo_angle) {
147 servo_set_angle(cmd.servo_angle);
148 last_servo_angle = cmd.servo_angle;
149 }
150
151 if (cmd.mezclar && !motor_activo) {
152
153 ESP_LOGI(TAG, "Iniciando mezcla");
154
155 gpio_set_level(PIN_MOTOR, 1);
156
157 motor_activo = true;
158 motor_start_time = xTaskGetTickCount();
159 }
160 }
161
162 if (motor_activo) {
163
164 TickType_t now = xTaskGetTickCount();
165
166 if ((now - motor_start_time) >= pdMS_TO_TICKS(MIX_DURATION_MS)) {
167
168 ESP_LOGI(TAG, "Deteniendo mezcla");
169
170 gpio_set_level(PIN_MOTOR, 0);
171 motor_activo = false;
172
173 xTaskNotifyGive(task_energia_handle);
174 }
175 }
176
177 vTaskDelay(pdMS_TO_TICKS(50));
178 }
179}
