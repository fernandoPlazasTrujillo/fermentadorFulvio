106static void wifi_event_handler(
107 void *arg,
108 esp_event_base_t event_base,
109 int32_t event_id,
110 void *event_data)
111{
115 if (event_base == WIFI_EVENT &&
116 event_id == WIFI_EVENT_STA_START)
117 {
118 ESP_LOGI(TAG, "Conectando al WiFi...");
119 retry_count = 0;
120 xEventGroupClearBits(
121 wifi_event_group,
122 WIFI_CONNECTED_BIT | WIFI_FAIL_BIT);
123 esp_wifi_connect();
124 }
125
129 else if (event_base == WIFI_EVENT &&
130 event_id == WIFI_EVENT_STA_DISCONNECTED)
131 {
132 wifi_event_sta_disconnected_t *event =
133 (wifi_event_sta_disconnected_t *) event_data;
134
135 xEventGroupClearBits(
136 wifi_event_group,
137 WIFI_CONNECTED_BIT);
138
139 if (retry_count < WIFI_MAX_RETRIES)
140 {
141 retry_count++;
142 ESP_LOGW(TAG,
143 "WiFi desconectado. Razon=%d. Reintento %d/%d...",
144 event->reason,
145 retry_count,
146 WIFI_MAX_RETRIES);
147 esp_wifi_connect();
148 }
149 else
150 {
151 ESP_LOGW(TAG,
152 "WiFi desconectado. Razon=%d. Sin mas reintentos.",
153 event->reason);
154 xEventGroupSetBits(
155 wifi_event_group,
156 WIFI_FAIL_BIT);
157 }
158 }
159
163 else if (event_base == IP_EVENT &&
164 event_id == IP_EVENT_STA_GOT_IP)
165 {
166 ip_event_got_ip_t *event =
167 (ip_event_got_ip_t *) event_data;
168
169 ESP_LOGI(TAG,
170 "WiFi conectado");
171
172 ESP_LOGI(TAG,
173 "IP obtenida: " IPSTR,
174 IP2STR(&event->ip_info.ip));
175
176 xEventGroupSetBits(
177 wifi_event_group,
178 WIFI_CONNECTED_BIT);
179 xEventGroupClearBits(
180 wifi_event_group,
181 WIFI_FAIL_BIT);
182 retry_count = 0;
183 }
184}

199void wifi_init(void)
200{
204 wifi_event_group = xEventGroupCreate();
205
209 esp_err_t ret = nvs_flash_init();
210
211 if (ret == ESP_ERR_NVS_NO_FREE_PAGES ||
212 ret == ESP_ERR_NVS_NEW_VERSION_FOUND)
213 {
214 ESP_ERROR_CHECK(nvs_flash_erase());
215
216 ret = nvs_flash_init();
217 }
218
219 ESP_ERROR_CHECK(ret);
220
224 ESP_ERROR_CHECK(esp_netif_init());
225
229 ESP_ERROR_CHECK(
230 esp_event_loop_create_default());
231
235 esp_netif_create_default_wifi_sta();
236
240 wifi_init_config_t cfg =
241 WIFI_INIT_CONFIG_DEFAULT();
242
243 ESP_ERROR_CHECK(esp_wifi_init(&cfg));
244
248 ESP_ERROR_CHECK(
249 esp_event_handler_instance_register(
250 WIFI_EVENT,
251 ESP_EVENT_ANY_ID,
252 &wifi_event_handler,
253 NULL,
254 NULL));
255
256 ESP_ERROR_CHECK(
257 esp_event_handler_instance_register(
258 IP_EVENT,
259 IP_EVENT_STA_GOT_IP,
260 &wifi_event_handler,
261 NULL,
262 NULL));
263
267 wifi_config_t wifi_config = {
268 .sta = {
269 .ssid = WIFI_SSID,
270 .password = WIFI_PASS,
271 },
272 };
273
277 ESP_ERROR_CHECK(
278 esp_wifi_set_mode(WIFI_MODE_STA));
279
283 ESP_ERROR_CHECK(
284 esp_wifi_set_config(
285 WIFI_IF_STA,
286 &wifi_config));
287
291 ESP_ERROR_CHECK(esp_wifi_start());
292
293 ESP_LOGI(TAG,
294 "Inicialización WiFi completada");
295}

307bool wifi_wait_connected(uint32_t timeout_ms)
308{
309 if (wifi_event_group == NULL)
310 {
311 return false;
312 }
313
314 EventBits_t bits = xEventGroupWaitBits(
315 wifi_event_group,
316 WIFI_CONNECTED_BIT | WIFI_FAIL_BIT,
317 pdFALSE,
318 pdFALSE,
319 pdMS_TO_TICKS(timeout_ms));
320
321 return (bits & WIFI_CONNECTED_BIT) != 0;
322}

334static const char *auth_mode_name(wifi_auth_mode_t authmode)

374void wifi_scan_print(void)
375{
376 wifi_ap_record_t ap_records[WIFI_SCAN_MAX_AP];
377 uint16_t ap_count = WIFI_SCAN_MAX_AP;
378
379 wifi_scan_config_t scan_config = {
380 .ssid = NULL,
381 .bssid = NULL,
382 .channel = 0,
383 .show_hidden = true
384 };
385
386 ESP_LOGI(TAG, "Escaneando redes WiFi visibles...");
387
388 esp_wifi_disconnect();
389 vTaskDelay(pdMS_TO_TICKS(500));
390
391 esp_err_t ret = esp_wifi_scan_start(&scan_config, true);
392 if (ret != ESP_OK)
393 {
394 ESP_LOGE(TAG, "Error iniciando escaneo WiFi: %s", esp_err_to_name(ret));
395 return;
396 }
397
398 ret = esp_wifi_scan_get_ap_records(&ap_count, ap_records);
399 if (ret != ESP_OK)
400 {
401 ESP_LOGE(TAG, "Error leyendo escaneo WiFi: %s", esp_err_to_name(ret));
402 return;
403 }
404
405 ESP_LOGI(TAG, "Redes encontradas: %d", ap_count);
406
407 for (int i = 0; i < ap_count; i++)
408 {
409 ESP_LOGI(TAG,
410 "%02d SSID='%s' RSSI=%d canal=%d auth=%s",
411 i + 1,
412 (char *) ap_records[i].ssid,
413 ap_records[i].rssi,
414 ap_records[i].primary,
415 auth_mode_name(ap_records[i].authmode));
416 }
417}
