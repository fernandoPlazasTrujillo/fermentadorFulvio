98static void mqtt_event_handler(void *handler_args,
99 esp_event_base_t base,
100 int32_t event_id,
101 void *event_data)
102{
103 esp_mqtt_event_handle_t event = event_data;
104
105 switch ((esp_mqtt_event_id_t) event_id)
106 {
107 case MQTT_EVENT_CONNECTED:
108 ESP_LOGI(TAG, "MQTT conectado");
109
110 mqtt_connected = true;
111
112 xEventGroupSetBits(mqtt_event_group, MQTT_CONNECTED_BIT);
113 xEventGroupClearBits(mqtt_event_group, MQTT_ERROR_BIT);
114 break;
115
116 case MQTT_EVENT_DISCONNECTED:
117 ESP_LOGW(TAG, "MQTT desconectado");
118
119 mqtt_connected = false;
120
121 xEventGroupClearBits(mqtt_event_group, MQTT_CONNECTED_BIT);
122 break;
123
124 case MQTT_EVENT_PUBLISHED:
125 ESP_LOGI(TAG, "MQTT publicado, msg_id=%d", event->msg_id);
126 xEventGroupSetBits(mqtt_event_group, MQTT_PUBLISHED_BIT);
127 break;
128
129 case MQTT_EVENT_ERROR:
130 ESP_LOGE(TAG, "Error MQTT");
131 xEventGroupSetBits(mqtt_event_group, MQTT_ERROR_BIT);
132 break;
133
134 default:
135 break;
136 }
137}

145void mqtt_manager_start(void)
146{
147 if (mqtt_event_group == NULL)
148 {
149 mqtt_event_group = xEventGroupCreate();
150 }
151
152 if (mqtt_event_group == NULL)
153 {
154 ESP_LOGE(TAG, "No se pudo crear grupo de eventos MQTT");
155 return;
156 }
157
158 xEventGroupClearBits(mqtt_event_group,
159 MQTT_CONNECTED_BIT | MQTT_PUBLISHED_BIT | MQTT_ERROR_BIT);
160
161 if (mqtt_client != NULL)
162 {
163 return;
164 }
165
166 esp_mqtt_client_config_t mqtt_cfg = {
167 .broker.address.uri = MQTT_BROKER_URI,
168 };
169
170 mqtt_client = esp_mqtt_client_init(&mqtt_cfg);
171 if (mqtt_client == NULL)
172 {
173 ESP_LOGE(TAG, "No se pudo crear cliente MQTT");
174 return;
175 }
176
177 esp_mqtt_client_register_event(
178 mqtt_client,
179 ESP_EVENT_ANY_ID,
180 mqtt_event_handler,
181 NULL);
182
183 ESP_LOGI(TAG, "Conectando a broker MQTT: %s", MQTT_BROKER_URI);
184 esp_mqtt_client_start(mqtt_client);
185}

198bool mqtt_wait_connected(uint32_t timeout_ms)
199{
200 if (mqtt_event_group == NULL)
201 {
202 return false;
203 }
204
205 EventBits_t bits = xEventGroupWaitBits(
206 mqtt_event_group,
207 MQTT_CONNECTED_BIT | MQTT_ERROR_BIT,
208 pdFALSE,
209 pdFALSE,
210 pdMS_TO_TICKS(timeout_ms));
211
212 return (bits & MQTT_CONNECTED_BIT) != 0;

232bool mqtt_publish(const char *topic, const char *payload, int qos, uint32_t timeout_ms)
233{
234 if (mqtt_client == NULL || mqtt_event_group == NULL)

236 return false;
237 }
238
239 xEventGroupClearBits(mqtt_event_group, MQTT_PUBLISHED_BIT | MQTT_ERROR_BIT);
240
241 int msg_id = esp_mqtt_client_publish(
242 mqtt_client,
243 topic,
244 payload,
245 0,
246 qos,
247 0);
248
249 if (msg_id < 0)
250 {
251 ESP_LOGE(TAG, "No se pudo publicar en MQTT");
252 return false;
253 }
254
255 if (qos == 0)
256 {
257 return true;
258 }
259

261 mqtt_event_group,
262 MQTT_PUBLISHED_BIT | MQTT_ERROR_BIT,
263 pdFALSE,
264 pdFALSE,
265 pdMS_TO_TICKS(timeout_ms));
266
267 return (bits & MQTT_PUBLISHED_BIT) != 0;
268}
