115static bool resend_pending_mqtt_messages(void)
116{
117 char pending_payload[MQTT_PENDING_PAYLOAD_SIZE];
118 FILE *pending_file;
119
120 if (!sd_card_pending_mqtt_exists())
121 {
122 return true;
123 }
124
125 printf("Reenviando mensajes MQTT pendientes desde SD\n");
126
127 pending_file = sd_card_open_pending_mqtt_read();
128
129 if (pending_file == NULL)
130 {
131 return false;
132 }
133
134 while (sd_card_read_pending_mqtt_line(
135 pending_file,
136 pending_payload,
137 sizeof(pending_payload)) != NULL)
138 {
139 pending_payload[strcspn(
140 pending_payload,
141 "\r\n")] = '\0';
142
143 if (pending_payload[0] == '\0')
144 {
145 continue;
146 }
147
148 printf("MQTT pendiente -> %s\n", pending_payload);
149
150 if (!mqtt_publish(

152 pending_payload,
153 1,
154 MQTT_PUBLISH_TIMEOUT_MS))
155 {
156 printf("Error reenviando payload MQTT pendiente\n");
157
158 sd_card_close_pending_mqtt(
159 pending_file);
160
161 return false;
162 }
163 }
164
165 sd_card_close_pending_mqtt(
166 pending_file);
167
168 if (sd_card_delete_pending_mqtt() != ESP_OK)
169 {
170 printf("Mensajes pendientes reenviados, pero no se pudo eliminar el archivo\n");
171 return false;

231void task_mqtt(void *pvParameters)
232{
233 sensor_data_t data;
234 char payload[MQTT_PENDING_PAYLOAD_SIZE];
235
236 while (1)
237 {
238 if (xQueueReceive(
239 queue_mqtt,
240 &data,
241 portMAX_DELAY))
242 {
243 build_payload(
244 &data,
245 payload,
246 sizeof(payload));
247
248 printf(
249 "MQTT payload -> %s\n",
250 payload);
251
252 mqtt_manager_start();
253
254 if (!mqtt_wait_connected(
255 MQTT_CONNECT_TIMEOUT_MS))
256 {
257 printf("MQTT no conectado\n");
258
259 save_pending_payload(payload);
260
261 mqtt_manager_stop();
262
263 xTaskNotifyGive(
264 task_energia_handle);
265
266 continue;
267 }
268
269 if (!resend_pending_mqtt_messages())
270 {
271 printf("No se pudieron reenviar todos los pendientes MQTT\n");
272
273 save_pending_payload(payload);
274
275 mqtt_manager_stop();
276
277 xTaskNotifyGive(
278 task_energia_handle);
279
280 continue;
281 }
282
283 if (mqtt_publish(
284 MQTT_TOPIC,
285 payload,
286 1,
287 MQTT_PUBLISH_TIMEOUT_MS))
288 {
289 printf("Dato publicado por MQTT\n");
290 }
291 else
292 {
293 printf("Error publicando por MQTT");
294
295 save_pending_payload(payload);
296 }
297
298 mqtt_manager_stop();
299
300 xTaskNotifyGive(
301 task_energia_handle);
302 }
303 }
