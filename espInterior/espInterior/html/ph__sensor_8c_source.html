97float ph_read_voltage(void)
98{
99 float sum = 0;
100
101 for (int i = 0; i < NUM_SAMPLES; i++) {
102 sum += adc_manager_read_voltage(PH_ADC_CHANNEL);
103
104 /*
105 * Se introduce un pequeño retardo entre muestras
106 * para evitar lecturas consecutivas excesivamente
107 * correlacionadas y mejorar el promedio obtenido.
108 */
109 vTaskDelay(pdMS_TO_TICKS(10));
110 }
111
112 return sum / NUM_SAMPLES;
113}

134float ph_read(void)
135{
136 float voltage = ph_read_voltage();
137
138 if (voltage > PH_MAX_VOLTAGE ||
139 voltage < PH_MIN_VOLTAGE)
140 {
141 ESP_LOGW(TAG, "Sensor pH fuera de rango");
142 return -1.0;
143 }
144
145 /*
146 * Conversión lineal obtenida durante el proceso
147 * de calibración del sensor.
148 */
149 return PH_SLOPE * voltage + PH_OFFSET;
150}
