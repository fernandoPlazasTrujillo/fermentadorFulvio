112esp_err_t sd_card_init(void)
113{
114 esp_err_t ret;
115
116 if (sd_mounted) {
117 return ESP_OK;
118 }
119
120 if (spi_mutex == NULL) {
121 spi_mutex = xSemaphoreCreateMutex();
122 if (spi_mutex == NULL) {
123 ESP_LOGE(TAG, "Error creando mutex SPI");
124 return ESP_FAIL;
125 }
126 }
127
128 sdmmc_host_t host = SDSPI_HOST_DEFAULT();
129
130 spi_bus_config_t bus_cfg = {
131 .mosi_io_num = PIN_MOSI,
132 .miso_io_num = PIN_MISO,
133 .sclk_io_num = PIN_CLK,
134 .quadwp_io_num = -1,
135 .quadhd_io_num = -1,
136 .max_transfer_sz = 4000,
137 };
138
139 /*
140 * Configuración del bus SPI utilizado para la comunicación
141 * con la tarjeta SD.
142 */
143 ret = spi_bus_initialize(host.slot, &bus_cfg, SDSPI_DEFAULT_DMA);
144 if (ret != ESP_OK && ret != ESP_ERR_INVALID_STATE) {
145 ESP_LOGE(TAG, "Error inicializando SPI");
146 return ret;
147 }
148
149 sdspi_device_config_t slot_config = SDSPI_DEVICE_CONFIG_DEFAULT();
150 slot_config.gpio_cs = PIN_CS;
151 slot_config.host_id = host.slot;
152
153 /*
154 * Montaje del sistema de archivos FAT sobre la tarjeta SD.
155 */
156 esp_vfs_fat_sdmmc_mount_config_t mount_config = {
157 .format_if_mount_failed = false,
158 .max_files = 5,
159 .allocation_unit_size = 16 * 1024
160 };
161
162 sdmmc_card_t *card;
163
164 ret = esp_vfs_fat_sdspi_mount(mount_point, &host, &slot_config, &mount_config, &card);
165
166 if (ret != ESP_OK) {
167 ESP_LOGE(TAG, "Error montando SD (%s)", esp_err_to_name(ret));
168 return ret;
169 }
170
171 ESP_LOGI(TAG, "SD montada correctamente");
172
173 sd_mounted = true;
174
175 return ESP_OK;
176}

190esp_err_t sd_card_write_line(const char *line)
191{
192 if (line == NULL) {
193 return ESP_FAIL;
194 }
195
196 if (spi_mutex != NULL) {
197 xSemaphoreTake(spi_mutex, portMAX_DELAY);
198 }
199
200 /*
201 * Apertura del archivo en modo append para preservar
202 * los registros históricos previamente almacenados.
203 */
204 FILE *f = fopen("/sdcard/log.csv", "a");
205
206 if (f == NULL) {
207 ESP_LOGE(TAG, "Error abriendo archivo");
208
209 if (spi_mutex != NULL) {
210 xSemaphoreGive(spi_mutex);
211 }
212
213 return ESP_FAIL;
214 }
215
216 /*
217 * Cada registro corresponde a una muestra del sistema
218 * de fermentación almacenada en formato CSV.
219 */
220 fprintf(f, "%s\n", line);
221
222 fflush(f);
223 fclose(f);
224
225 if (spi_mutex != NULL) {
226 xSemaphoreGive(spi_mutex);
227 }
228
229 return ESP_OK;
230}

232esp_err_t sd_card_append_pending_mqtt(const char *payload)
233{
234 if (payload == NULL) {
235 return ESP_FAIL;
236 }
237
238 if (sd_card_init() != ESP_OK) {
239 return ESP_FAIL;
240 }
241
242 if (spi_mutex != NULL) {
243 xSemaphoreTake(spi_mutex, portMAX_DELAY);
244 }
245
246 FILE *f = fopen(pending_mqtt_path, "a");
247
248 if (f == NULL) {
249 ESP_LOGE(TAG, "Error abriendo pendientes MQTT (%s), errno=%d", pending_mqtt_path, errno);
250
251 if (spi_mutex != NULL) {
252 xSemaphoreGive(spi_mutex);
253 }
254
255 return ESP_FAIL;
256 }
257
258 fprintf(f, "%s\n", payload);
259 fflush(f);
260 fclose(f);
261
262 if (spi_mutex != NULL) {
263 xSemaphoreGive(spi_mutex);
264 }
265
266 return ESP_OK;
267}

323esp_err_t sd_card_delete_pending_mqtt(void)
324{
325 if (sd_card_init() != ESP_OK) {
326 return ESP_FAIL;
327 }
328
329 if (spi_mutex != NULL) {
330 xSemaphoreTake(spi_mutex, portMAX_DELAY);
331 }
332
333 int result = remove(pending_mqtt_path);
334
335 if (spi_mutex != NULL) {
336 xSemaphoreGive(spi_mutex);
337 }
338
339 if (result == 0 || errno == ENOENT) {
340 return ESP_OK;
341 }
342
343 ESP_LOGE(TAG, "Error eliminando pendientes MQTT (%s), errno=%d", pending_mqtt_path, errno);
344 return ESP_FAIL;
345}
