214esp_err_t oled_draw_pixel(uint8_t x, uint8_t y)
215{
216 if (x >= OLED_WIDTH)
217 return ESP_FAIL;
218
219 if (y >= OLED_HEIGHT)
220 return ESP_FAIL;
221
222 framebuffer[
223 x + (y / 8) * OLED_WIDTH
224 ] |= (1 << (y % 8));
225
226 return ESP_OK;
227}

230esp_err_t oled_draw_char(uint8_t x, uint8_t y, char c)
231{
232 if ((uint8_t)c >= sizeof(font5x7)/5)
233 return ESP_FAIL;
234
235 for (int col = 0; col < 5; col++)
236 {
237 uint8_t line = font5x7[(uint8_t)c][col];
238
239 for (int row = 0; row < 7; row++)
240 {
241 if (line & (1 << row))
242 {
243 oled_draw_pixel(
244 x + col,
245 y + row);
246 }
247 }
248 }
249
250 return ESP_OK;
251}

272esp_err_t oled_update(void)
273{

276 page++)
