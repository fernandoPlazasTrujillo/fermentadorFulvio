219float ds18b20_read_temperature(void)
220{
221 if (!ow_reset()) {
222 ESP_LOGE(TAG, "Sensor no detectado");
223 return -127.0;
224 }
225
226 ow_write_byte(0xCC);
227 ow_write_byte(0x44);
228
229 /*
230 * Tiempo máximo de conversión para resolución de 12 bits
231 * según la hoja de datos del DS18B20.
232 */
233 vTaskDelay(pdMS_TO_TICKS(750));
234
235 if (!ow_reset()) {
236 return -127.0;
237 }
238
239 ow_write_byte(0xCC);
240 ow_write_byte(0xBE);
241
242 uint8_t temp_lsb = ow_read_byte();
243 uint8_t temp_msb = ow_read_byte();
244
245 int16_t raw = (temp_msb << 8) | temp_lsb;
246
247 float temp = raw / 16.0;
248
249 return temp;
250}
