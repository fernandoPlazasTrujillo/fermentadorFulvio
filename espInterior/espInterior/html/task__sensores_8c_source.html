116void task_sensores(void *pvParameters)
117{
118 sensor_data_t data;
119
120 // =====================================================
121 // INICIALIZACIÓN DE DRIVERS
122 // =====================================================
123
124 mq135_init();
125
126 ph_init();
127
128 ds18b20_init(GPIO_NUM_4);
129
130 while (1)
131 {
132 // =====================================================
133 // ESPERAR ACTIVACIÓN
134 // =====================================================
135
136 ulTaskNotifyTake(
137 pdTRUE,
138 portMAX_DELAY);
139
140 ESP_LOGI(TAG, "Leyendo sensores...");
141
142 // =====================================================
143 // SENSOR MQ135
144 // =====================================================
145
146 float raw_voltage =
147 mq135_read();
148
149 int raw_adc =
150 mq135_read_raw();
151
152 ESP_LOGI(
153 TAG,
154 "MQ135 ADC RAW: %d",
155 raw_adc);
156
161 filtered =
162 ALPHA * raw_voltage +
163 (1.0f - ALPHA) * filtered;
164
168 if (!baseline_initialized)
169 {
170 baseline = filtered;
171
172 baseline_initialized = 1;
173
174 ESP_LOGI(
175 TAG,
176 "Baseline inicial: %.4f",
177 baseline);
178 }
179
180 data.co2_raw = raw_voltage;
181
182 data.co2 =
183 filtered - baseline;
184
185 // =====================================================
186 // SENSOR DE PH
187 // =====================================================
188
189 data.ph =
190 ph_read();
191
192 // =====================================================
193 // TEMPERATURA DS18B20
194 // =====================================================
195
196 float temp =
197 ds18b20_read_temperature();
198
199 if (temp == -127.0f)
200 {
201 ESP_LOGW(
202 TAG,
203 "Error leyendo DS18B20");
204
205 temp = 25.0f;
206 }
207
208 data.temperatura = temp;
209
210 // =====================================================
211 // VARIABLES DE PRUEBA
212 // =====================================================
213
214 data.voltaje = 5.0f;
215
216 data.corriente = 0.5f;
217
218 // =====================================================
219 // RTC
220 // =====================================================
221
222 ds3231_get_datetime(
223 &data.datetime);
224
225 printf(
226 "RTC -> %02d:%02d:%02d\n",
227 data.datetime.hours,
228 data.datetime.minutes,
229 data.datetime.seconds);
230
231 // =====================================================
232 // ENVÍO A TASK CONTROL
233 // =====================================================
234
235 xQueueSend(
236 queue_sensores,
237 &data,
238 portMAX_DELAY);
239
240 // =====================================================
241 // ENVÍO A LOGGER
242 // =====================================================
243
244 xQueueSend(
245 queue_logger,
246 &data,
247 portMAX_DELAY);
248
249 // =====================================================
250 // ENVÍO A MQTT
251 // =====================================================
252
253 xQueueSend(
254 queue_mqtt,
255 &data,
256 portMAX_DELAY);
257
258 // =====================================================
259 // DATOS DE DISPLAY
260 // =====================================================
261
262 display_data_t display;
263
264 display.temperatura =
265 data.temperatura;
266
267 display.ph =
268 data.ph;
269
270 display.co2 =
271 data.co2;
272
273 display.wifi_ok =
274 wifi_is_connected();
275
276 display.mqtt_ok =
277 mqtt_is_connected();
278
279 display.sd_ok =
280 sd_card_is_mounted();
281
282 display.datetime =
283 data.datetime;
284
285 xQueueSend(
286 queue_display,
287 &display,
288 0);
289
290 ESP_LOGI(
291 TAG,
292 "Datos enviados");
293 }
