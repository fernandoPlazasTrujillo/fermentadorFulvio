93float voltage_sensor_read(void)
94{
95 float sum = 0.0f;
96
97 /*
98 * Promediado simple para reducir ruido.
99 */
100 for (int i = 0; i < NUM_SAMPLES; i++) {
101 sum += adc_manager_read_voltage(VOLTAGE_ADC_CHANNEL);
102 }
103
104 float v_adc = sum / NUM_SAMPLES;
105
106 /*
107 * Corrección de offset obtenida durante
108 * la calibración experimental.
109 */
110 v_adc -= ADC_OFFSET;
111
112 if (v_adc < 0.0f) {
113 v_adc = 0.0f;
114 }
115
116 /*
117 * Conversión del voltaje medido en el ADC
118 * al voltaje real mediante el divisor resistivo.
119 */
120 float v_real = v_adc * DIVIDER_RATIO;
121
122 ESP_LOGI(TAG, "Vadc corregido: %.3f V", v_adc);
123 ESP_LOGI(TAG, "Voltaje final: %.2f V", v_real);
124
125 return v_real;
126}
