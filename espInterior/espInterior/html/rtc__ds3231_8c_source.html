129esp_err_t ds3231_get_datetime(datetime_t *dt)
130{
131 uint8_t data[7];
132
133 if (xSemaphoreTake(mutex_i2c, pdMS_TO_TICKS(100)) != pdTRUE) {
134 return ESP_ERR_TIMEOUT;
135 }
136
137 i2c_manager_read(DS3231_ADDR, 0x00, data, 7);
138
139 xSemaphoreGive(mutex_i2c);
140
141 /*
142 * Conversión de registros DS3231 almacenados
143 * en formato BCD hacia formato decimal.
144 */
145 dt->seconds = bcd_to_dec(data[0]);
146 dt->minutes = bcd_to_dec(data[1]);
147 dt->hours = bcd_to_dec(data[2]);
148 dt->day = bcd_to_dec(data[4]);
149 dt->month = bcd_to_dec(data[5] & 0x1F);
150 dt->year = 2000 + bcd_to_dec(data[6]);
151
152 return ESP_OK;
153}

204esp_err_t ds3231_set_alarm_seconds(uint8_t seconds)
205{
206 uint8_t data[4];
207
208 /*
209 * Configuración de Alarm 1:
210 * - Coincidencia únicamente sobre segundos.
211 * - Minutos, horas y día ignorados mediante bits A1Mx.
212 */
213 data[0] = dec_to_bcd(seconds);
214 data[1] = 0x80;
215 data[2] = 0x80;
216 data[3] = 0x80;
217
218 if (xSemaphoreTake(mutex_i2c, pdMS_TO_TICKS(100)) != pdTRUE) {
219 return ESP_ERR_TIMEOUT;
220 }
221
222 i2c_manager_write(DS3231_ADDR, 0x07, data, 4);
223
224 /*
225 * INTCN = 1 -> modo interrupción
226 * A1IE = 1 -> habilita Alarm 1
227 */
228 uint8_t control = DS3231_CONTROL_A1_INTERRUPT;
229
230 i2c_manager_write(DS3231_ADDR, 0x0E, &control, 1);
231
232 xSemaphoreGive(mutex_i2c);
233
234 ESP_LOGI(TAG, "Alarma configurada en segundo: %d", seconds);
235
236 return ESP_OK;
237}
