155esp_err_t servo_set_angle(float angle)
156{
157 if (angle < 0.0f) {
158 angle = 0.0f;
159 }
160
161 if (angle > 180.0f) {
162 angle = 180.0f;
163 }
164
165 /*
166 * Conversión lineal de ángulo a ancho de pulso.
167 */
168 float pulse_us =
169 SERVO_MIN_PULSE_US +
170 ((angle / 180.0f) *
171 (SERVO_MAX_PULSE_US - SERVO_MIN_PULSE_US));
172
173 /*
174 * Conversión de ancho de pulso a duty cycle.
175 */
176 uint32_t duty =
177 (uint32_t)((pulse_us / SERVO_PERIOD_US) *
178 SERVO_MAX_DUTY);
179
180 ledc_set_duty(
181 SERVO_MODE,
182 SERVO_CHANNEL,
183 duty);
184
185 ledc_update_duty(
186 SERVO_MODE,
187 SERVO_CHANNEL);
188
189 ESP_LOGI(
190 TAG,
191 "Servo angle: %.2f deg | pulse: %.2f us | duty: %lu",
192 angle,
193 pulse_us,
194 (unsigned long)duty);
195
196 return ESP_OK;
197}
